clear;
clc;

while mod(n,2) ~=0
n = randi([10,20]);
end

A2 = 2 * ones(dim);

A4 = ones(dim);

det_A= det(A)
[V,D] = eig(A)
spettro = diag(D);
raggio_Spettrale = max(abs(spettro))
